## docker-compose.yml (DB 서버 생성)

```yaml
version: '3.8'
services:
  db:
    image: mysql:8.0                 # 사용할 이미지 (MySQL 8.0 버전)
    container_name: todo-mysql       # 컨테이너 이름
    environment:                     # 초기 계정 설정
      MYSQL_ROOT_PASSWORD: qwer    
      MYSQL_USER: todos-user         # 스프링 username과 일치해야 함
      MYSQL_PASSWORD: 1234           # 스프링 password와 일치해야 함
      MYSQL_DATABASE: todos          # 스프링 url의 DB 명과 일치해야 함
    ports:
      - "3306:3306"
    volumes:
      - ./db_data:/var/lib/mysql     # 데이터 영구 저장 위치 (컨테이너를 꺼도 유지되게) 
```

## application.yaml (DB 연결 정보)
```yaml
spring:
	datasource:
		# 접속 주소 3306 포트, DB 이름(todos)
    url: jdbc:mysql://localhost:3306/todos  
    
    # 계정 정보 (docker-compose.yml 정보와 일치해야 함)
    username: todos-user                    
    password: 1234

  jpa:
	  # 서버 실행 시, data.sql 실행 여부
    defer-datasource-initialization: true
    
    hibernate:
	    # 테이블 생성 전략 (update: 변경분만 반영, create: 매번 새로 만듦)
      ddl-auto: update
      
      # 실행되는 SQL을 콘솔에 보여줌
      show-sql: true
```
